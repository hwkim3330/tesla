Architecture Comparison

V11 - Modular Pipeline
# V11: Multiple separate components
class FSD_V11:
def __init__(self):
# Separate neural networks
self.detector = ObjectDetector()
self.lane_net = LaneDetector()
self.depth_net = DepthEstimator()
self.sign_net = SignClassifier()
# Rule-based planning (C++)
self.planner = RuleBasedPlanner()
self.controller = PIDController()
def forward(self, images):
# Step 1: Perception
objects = self.detector(images)
lanes = self.lane_net(images)
depth = self.depth_net(images)
# Step 2: Rule-based planning
trajectory = self.planner.plan(
objects, lanes, depth,
300000_lines_of_rules
)
# Step 3: Control
return self.controller(trajectory)
